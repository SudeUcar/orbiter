$$
\begin{array}{c}

\end{array}
$$}

X0^4 + X1^4 + X2^4 + X0^2*X1^2 + X0^2*X2^2 + X1^2*X2^2 + X0^2*X1*X2 + X0*X1^2*X2 + X0*X1*X2^2
\end{verbatim}
